```tsx
export type PageSizeValue = 10 | 20 | 30 | 40 | 50 | 100 | "all";

export const PAGE_SIZE_OPTIONS: PageSizeValue[] = [
  10,
  20,
  30,
  40,
  50,
  100,
  "all",
];

export function getPageSizeLabel(value: PageSizeValue) {
  return value === "all" ? "All" : String(value);
}

export function getShowingText(params: {
  page: number;
  pageSize: PageSizeValue;
  total: number;
  currentCount: number;
}) {
  const { page, pageSize, total, currentCount } = params;

  if (total === 0 || currentCount === 0) {
    return "Showing 0 to 0 of 0 records";
  }

  if (pageSize === "all") {
    return `Showing 1 to ${total} of ${total} records`;
  }

  const start = (page - 1) * pageSize + 1;
  const end = Math.min(start + currentCount - 1, total);

  return `Showing ${start} to ${end} of ${total} records`;
}

export function getTotalPages(params: {
  total: number;
  pageSize: PageSizeValue;
}) {
  const { total, pageSize } = params;

  if (pageSize === "all") return 1;
  if (total <= 0) return 1;

  return Math.max(1, Math.ceil(total / pageSize));
}
```

```tsx
"use client";

import { useEffect, useState } from "react";

export function useDebouncedValue<T>(value: T, delay = 400) {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const timerId = window.setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      window.clearTimeout(timerId);
    };
  }, [value, delay]);

  return debouncedValue;
}
```

```tsx
"use client";

import { getShowingText, getTotalPages, PageSizeValue } from "@/lib/pagination";

type EnterprisePaginationProps = {
  page: number;
  pageSize: PageSizeValue;
  total: number;
  currentCount: number;
  onPageChange: (page: number) => void;
};

function getPageNumbers(currentPage: number, totalPages: number) {
  const pages: number[] = [];

  const start = Math.max(1, currentPage - 2);
  const end = Math.min(totalPages, currentPage + 2);

  for (let page = start; page <= end; page += 1) {
    pages.push(page);
  }

  return pages;
}

export function EnterprisePagination({
  page,
  pageSize,
  total,
  currentCount,
  onPageChange,
}: EnterprisePaginationProps) {
  const totalPages = getTotalPages({ total, pageSize });
  const showingText = getShowingText({ page, pageSize, total, currentCount });

  const disabled = totalPages <= 1 || pageSize === "all";

  return (
    <div className="flex flex-col gap-3 border-t border-slate-100 px-4 py-4 text-sm md:flex-row md:items-center md:justify-between">
      <p className="text-slate-500">{showingText}</p>

      <div className="flex flex-wrap items-center gap-1">
        <button
          type="button"
          disabled={disabled || page === 1}
          onClick={() => onPageChange(1)}
          className="rounded-lg border border-slate-200 px-3 py-1.5 font-medium text-slate-600 transition hover:bg-slate-50 disabled:cursor-not-allowed disabled:opacity-40"
        >
          First
        </button>

        <button
          type="button"
          disabled={disabled || page === 1}
          onClick={() => onPageChange(page - 1)}
          className="rounded-lg border border-slate-200 px-3 py-1.5 font-medium text-slate-600 transition hover:bg-slate-50 disabled:cursor-not-allowed disabled:opacity-40"
        >
          Previous
        </button>

        {getPageNumbers(page, totalPages).map((pageNumber) => (
          <button
            type="button"
            key={pageNumber}
            onClick={() => onPageChange(pageNumber)}
            className={`rounded-lg border px-3 py-1.5 font-semibold transition ${
              pageNumber === page
                ? "border-slate-900 bg-slate-900 text-white"
                : "border-slate-200 text-slate-600 hover:bg-slate-50"
            }`}
          >
            {pageNumber}
          </button>
        ))}

        <button
          type="button"
          disabled={disabled || page === totalPages}
          onClick={() => onPageChange(page + 1)}
          className="rounded-lg border border-slate-200 px-3 py-1.5 font-medium text-slate-600 transition hover:bg-slate-50 disabled:cursor-not-allowed disabled:opacity-40"
        >
          Next
        </button>

        <button
          type="button"
          disabled={disabled || page === totalPages}
          onClick={() => onPageChange(totalPages)}
          className="rounded-lg border border-slate-200 px-3 py-1.5 font-medium text-slate-600 transition hover:bg-slate-50 disabled:cursor-not-allowed disabled:opacity-40"
        >
          Last
        </button>
      </div>
    </div>
  );
}
```

```tsx
import { ReactNode } from "react";
import { EmptyState } from "@/components/common/EmptyState";
import { EnterprisePagination } from "@/components/common/EnterprisePagination";
import { PageSizeValue } from "@/lib/pagination";

export type EnterpriseTableColumn<T> = {
  key: string;
  title: string;
  render: (row: T, index: number) => ReactNode;
  headerClassName?: string;
  cellClassName?: string;
  align?: "left" | "center" | "right";
};

type EnterpriseDataTableProps<T> = {
  rows: T[];
  columns: EnterpriseTableColumn<T>[];
  getRowKey: (row: T, index: number) => string | number;

  loading?: boolean;
  error?: string | null;

  emptyTitle?: string;
  emptyMessage?: string;
  emptyAction?: ReactNode;

  page: number;
  pageSize: PageSizeValue;
  total: number;
  onPageChange: (page: number) => void;
};

function getAlignClass(align?: "left" | "center" | "right") {
  if (align === "center") return "text-center";
  if (align === "right") return "text-right";

  return "text-left";
}

function TableSkeleton({ columnCount }: { columnCount: number }) {
  return (
    <>
      {Array.from({ length: 5 }).map((_, rowIndex) => (
        <tr key={rowIndex} className="border-t border-slate-100">
          {Array.from({ length: columnCount }).map((__, columnIndex) => (
            <td key={columnIndex} className="px-4 py-4">
              <div className="h-4 w-full animate-pulse rounded bg-slate-100" />
            </td>
          ))}
        </tr>
      ))}
    </>
  );
}

export function EnterpriseDataTable<T>({
  rows,
  columns,
  getRowKey,
  loading = false,
  error = null,
  emptyTitle = "No records found",
  emptyMessage = "Try changing your search or filter options.",
  emptyAction,
  page,
  pageSize,
  total,
  onPageChange,
}: EnterpriseDataTableProps<T>) {
  const showEmpty = !loading && !error && rows.length === 0;
  const showTable = !showEmpty && !error;

  return (
    <div className="overflow-hidden rounded-2xl border border-slate-200 bg-white shadow-sm">
      {error ? (
        <div className="p-6">
          <div className="rounded-2xl border border-rose-200 bg-rose-50 px-4 py-3 text-sm font-medium text-rose-700">
            {error}
          </div>
        </div>
      ) : null}

      {showEmpty ? (
        <div className="p-4">
          <EmptyState
            title={emptyTitle}
            message={emptyMessage}
            action={emptyAction}
          />
        </div>
      ) : null}

      {showTable ? (
        <div className="overflow-x-auto">
          <table className="min-w-full divide-y divide-slate-100">
            <thead className="bg-slate-50">
              <tr>
                {columns.map((column) => (
                  <th
                    key={column.key}
                    className={`whitespace-nowrap px-4 py-3 text-xs font-bold uppercase tracking-wide text-slate-500 ${getAlignClass(
                      column.align
                    )} ${column.headerClassName || ""}`}
                  >
                    {column.title}
                  </th>
                ))}
              </tr>
            </thead>

            <tbody className="divide-y divide-slate-100 bg-white">
              {loading ? (
                <TableSkeleton columnCount={columns.length} />
              ) : (
                rows.map((row, index) => (
                  <tr
                    key={getRowKey(row, index)}
                    className="transition hover:bg-slate-50/70"
                  >
                    {columns.map((column) => (
                      <td
                        key={column.key}
                        className={`whitespace-nowrap px-4 py-3 text-sm text-slate-700 ${getAlignClass(
                          column.align
                        )} ${column.cellClassName || ""}`}
                      >
                        {column.render(row, index)}
                      </td>
                    ))}
                  </tr>
                ))
              )}
            </tbody>
          </table>
        </div>
      ) : null}

      <EnterprisePagination
        page={page}
        pageSize={pageSize}
        total={total}
        currentCount={rows.length}
        onPageChange={onPageChange}
      />
    </div>
  );
}
```
